SELECT *
FROM (SELECT servedCar, count(servedCar) AS occurrence
      FROM RentHistory
      WHERE date >= '2018-10-01'
      GROUP BY servedCar
      ORDER BY occurrence) AS OrdersOccurence
LIMIT (SELECT (COUNT(*) * 0.1) :: integer
       FROM (SELECT servedCar, count(servedCar) AS occurrence
             FROM RentHistory
             WHERE date >= '2018-10-01'
             GROUP BY servedCar
             ORDER BY occurrence) AS OrdersOccurence);
