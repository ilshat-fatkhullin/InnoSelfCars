SELECT pickUpLocation
FROM (SELECT pickUpLocation, count(pickUpLocation) AS occurrence
      FROM RentHistory
      WHERE time >= '7:00'
        AND time <= '10:00'
      GROUP BY pickUpLocation
      ORDER BY occurrence DESC
      LIMIT 3) AS Morning;
